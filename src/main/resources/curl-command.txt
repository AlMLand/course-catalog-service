================
CourseController
================
CREATE -> createCourse :
curl -i -d '{"id":null, "name":"Kotlin", "category": [{"category":"DEVELOPMENT", "id":"that is X" ,"description":"some value1"}], "instructorId":{"firstName":"Alex", "lastName":"M_land"}}' -H "Content-Type: application/json" -X POST http://localhost:8080/v1/courses
curl -i -d '{"id":null, "name":"Spring Boot", "category": [{"category":"DEVELOPMENT", "id":"that is X" ,"description":"some value2"}], "instructorId":{"firstName":"Timur", "lastName":"M_land"}}' -H "Content-Type: application/json" -X POST http://localhost:8080/v1/courses
curl -i -d '{"id":null, "name":"React", "category": [{"category":"DEVELOPMENT", "id":"that is X" ,"description":"some value3"}], "instructorId":{"firstName":"Timur", "lastName":"M_land"}}' -H "Content-Type: application/json" -X POST http://localhost:8080/v1/courses
curl -i -d '{"id":null, "name":"Angular", "category": [{"category":"DEVELOPMENT", "id":"that is X" ,"description":"some value3"}, {"category":"MANAGEMENT", "id":"that is Y" ,"description":"some value4"}], "instructorId":{"firstName":"Alex", "lastName":"M_land"}}' -H "Content-Type: application/json" -X POST http://localhost:8080/v1/courses

GET -> getCourse :
curl -i http://localhost:8080/v1/courses/1

GET -> getAllCourses :
curl -i http://localhost:8080/v1/courses

GET -> getAllCourses with parameter name :
curl -i http://localhost:8080/v1/courses?name=Kotlin

GET -> getAllCourses with parameter category :
curl -i http://localhost:8080/v1/courses?category=DEVELOPMENT

GET -> getAllCourses with parameters name, category  :
curl -i http://localhost:8080/v1/courses?name=Kotlin&category=DEVELOPMENT

UPDATE -> updateCourse :
curl -i -d '{"id":null, "name":"Java", "category":[{"category":"MANAGEMENT", "id":"that id is Y", "description":"some test management"}], "instructorId":{"firstName":"Timur", "lastName":"M_land"}}' -H "Content-Type: application/json" -X PUT http://localhost:8080/v1/courses/1

DELETE -> deleteCourse :
curl -i -X DELETE http://localhost:8080/v1/courses/1
